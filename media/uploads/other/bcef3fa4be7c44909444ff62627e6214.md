### 3.2 йҖҶеҗ‘з”ҹжҲҗ

```bash
python manage.py inspectdb --database=default > scripts/_generated_models.py
```

з„¶еҗҺжүӢеҠЁж•ҙзҗҶдёәеҗ„ app дёӢзҡ„ models.pyгҖӮиЎЁеҗҚе’Ңеӯ—ж®өеҗҚдҝқжҢҒеҺҹж ·пјҲдёҚз”Ё Django й»ҳи®Өзҡ„ `appname_modelname`пјүпјҢйҖҡиҝҮ `db_table` жҢҮе®ҡеҺҹиЎЁеҗҚпјҡ

```python
class TbUser(models.Model):
    id = models.CharField(primary_key=True, max_length=50)
    id_card = models.CharField(max_length=50, blank=True, null=True)
    school_id = models.CharField(max_length=50, blank=True, null=True)
    specialty_id = models.CharField(max_length=50, blank=True, null=True)
    class_id = models.CharField(max_length=50, blank=True, null=True)
    name = models.CharField(max_length=50, blank=True, null=True)
    username = models.CharField(max_length=50, blank=True, null=True)
    password = models.CharField(max_length=50, blank=True, null=True)
    type = models.IntegerField(blank=True, null=True)  # 0:еӯҰз”ҹ 1:ж•ҷеёҲ 2:з®ЎзҗҶе‘ҳ
    # ... е…¶д»–еӯ—ж®ө
    create_time = models.DateTimeField(blank=True, null=True)
    expire_time = models.DateTimeField(blank=True, null=True)

    class Meta:
        managed = False
        db_table = 'tb_user'
```

    async def connect(self):
        self.experiment_id = self.scope['url_route']['kwargs']['experiment_id']
        await self.channel_layer.group_add(
            f"experiment_{self.experiment_id}",
            self.channel_name
        )
        await self.accept()

    async def receive(self, text_data):
        data = json.loads(text_data)
        # еӨ„зҗҶе®һж—¶е®һйӘҢж•°жҚ®
        await self.channel_layer.group_send(
            f"experiment_{self.scope['url_route']['kwargs']['experiment_id']}",
            {'type': 'experiment.update', 'data': data}
        )

    async def experiment_update(self, event):
        await self.send(text_data=json.dumps(event['data']))
